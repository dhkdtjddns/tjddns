 public static void main(String[] args) {
        int n[][] = {{1, 2, 3},{1, 2}, {1}, {1, 2, 3}, {1, 2, 3, 4}};

        // 2차원 배열의 각 행을 출력
        for (int i = 0; i < n.length; i++) {
            for (int j = 0; j < n[i].length; j++) {
                System.out.print(n[i][j] + "\t"); // 각 요소를 탭으로 구분하여 출력
            }
            System.out.println(); // 각 행의 끝에서 줄바꿈
        }
    }
